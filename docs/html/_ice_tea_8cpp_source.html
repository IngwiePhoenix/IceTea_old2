448 int IceTea::run() {
449  // Return code tracker
450  int rt = 0;
451 
452  // Startup notice
453  stringstream ss;
454  ss << "Starting up. CWD: "
455  << folder_current_full()
456  << " : "
457  << folder_current();
458  this->printDebug(ss.str());
459 
460  // Setup everything.
461  if(!this->startup()) {
462  this->printDebug("Startup failed.");
463  return -1;
464  }
465 
466  // Initialize all the targets, rules, actions, and possibly more.
467  this->printDebug("Calling IceTea.Initializer...");
468  callObjectFunction(this, "IceTea", "Initializer", 0, 0, false);
469  if(this->hasEndedExecuting(rt)) {
470  this->handleException();
471  this->printDebug("Initializer failed.");
472  return rt;
473  } else {
474  rt = this->popNumber();
475  }
476 
477  if(this->checkAndRunHelp()) {
478  // We ran the help. Exit with 0.
479  return 0;
480  } if(this->checkAndRunInline(rt)) {
481  // We ran an inline script, return.
482  return rt;
483  } if(this->checkAndListModules(rt)) {
484  return rt;
485  }
486 
487  // Is there anything else?
488  performIceTeaEvent(this, "postprocessCli", 1);
489  if(this->hasEndedExecuting(rt)) {
490  this->printDebug("event:postprocessCli failed.");
491  return rt;
492  }
493 
494  bool shouldContinue = this->isType(OS_VALUE_TYPE_BOOL) ? this->popBool() : false;
495  if(!shouldContinue) {
496  // We did something in postprocessCli, so we should exit.
497  this->printDebug("Closing after event:postprocessCli");
498  return 0;
499  }
500 
501  // If we came thus far, then we really want to configure, stack and run.
502  callObjectFunction(this, "IceTea", "Main");
503  if(this->hasEndedExecuting(rt)) {
504  this->printDebug("Main failed.");
505  return rt;
506  } else {
507  rt = this->popNumber();
508  }
509 
510  return rt;
511 }
