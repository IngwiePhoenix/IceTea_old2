182  strayArgs.clear();
183  int argn;
184  std::pair<std::string, int> opt;
185  for(argn=1; argn < argc; argn++) { /* Read command line. */
186  opt = findOpt(argv[argn]);
187  if(opt.second!=-1) { /* This argument is an option switch. */
188  if(ga.find(opt.first)!=ga.end() && !ga[opt.first][opt.second]->arg.empty()) { /* It takes an argument. */
189  if(argn + 1 < argc) {
190  ga[opt.first][opt.second]->value = argv[argn+1]; /* Add next argument as the option value. */
191  argn++;
192  } else return false;
193  }
194  ga[opt.first][opt.second]->present = true; /* Tell the option it exists. */
195  } else if(argv[argn][0] == '-') /* Nonexistent option. */
