\subsection{Detailed Description}
Condition variable class. 

This is a signalling object for synchronizing the execution flow for several threads. Example usage\+: 
\begin{DoxyCode}
\textcolor{comment}{// Shared data and associated mutex and condition variable objects}
\textcolor{keywordtype}{int} count;
mutex m;
\hyperlink{classtthread_1_1condition__variable_ab7d3340f1fe6fbe922a2534762312d76}{condition\_variable} cond;

\textcolor{comment}{// Wait for the counter to reach a certain number}
\textcolor{keywordtype}{void} wait\_counter(\textcolor{keywordtype}{int} targetCount)
\{
  lock\_guard<mutex> guard(m);
  \textcolor{keywordflow}{while}(count < targetCount)
    cond.wait(m);
\}

\textcolor{comment}{// Increment the counter, and notify waiting threads}
\textcolor{keywordtype}{void} increment()
\{
  lock\_guard<mutex> guard(m);
  ++ count;
  cond.notify\_all();
\}
\end{DoxyCode}
